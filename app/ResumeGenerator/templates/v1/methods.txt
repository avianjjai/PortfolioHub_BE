% Sections formatting for ATS compatibility
\titleformat{\section}{
    \vspace{-3pt}  % Reduce excess spacing for cleaner layout
    \scshape\raggedright\large\bfseries
}{}{0em}{}[\color{black}
\titlerule
\vspace{-4pt}  % Reduced spacing to prevent ATS issues
]

% Resume item with consistent spacing
\newcommand{\resumeItem}[1]{
    \item {
        \small{#1\vspace{-2pt}}
    }
}

% Classes list for courses or skills
\newcommand{\classesList}[4]{
    \item {
        \small{#1, #2, #3, #4\vspace{-2pt}}
    }
}

% Resume subheading with position, company, date, and location
\newcommand{\resumeSubheading}[4]{
    \vspace{-2pt}
    \item {
        \begin{tabular*}{1.0\textwidth}[t]{l@{\extracolsep{\fill}}r}
            \textbf{\large #1} & \textbf{\small #2} \\
            \textit{\large #3} & \textit{\small #4} \\
        \end{tabular*}
        \vspace{-6pt}
    }
}

% Sub-subheading for additional info under subheading
\newcommand{\resumeSubSubheading}[2]{
    \item {
        \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
            \textit{\small #1} & \textit{\small #2} \\
        \end{tabular*}\vspace{-6pt}
    }
}

% Project heading for projects with title and date
\newcommand{\resumeProjectHeading}[2]{
    \item {
        \begin{tabular*}{1.001\textwidth}{l@{\extracolsep{\fill}}r}
            \small #1 & \textbf{\small #2} \\
        \end{tabular*}
        \vspace{-12pt}  % Reduced spacing for better alignment
    }
}

% Sub item under a heading or project
\newcommand{\resumeSubItem}[1]{\resumeItem{#1}\vspace{-3pt}}

% Standardizing bullet points for ATS
\renewcommand\labelitemi{$\bullet$}
\renewcommand\labelitemii{$\bullet$}

% Consistent list start and end
\newcommand{\resumeSubHeadingListStart}{\begin{itemize}[leftmargin=0.0in, label={}]}
\newcommand{\resumeSubHeadingListEnd}{\end{itemize}}
\newcommand{\resumeItemListStart}{\begin{itemize}}
\newcommand{\resumeItemListEnd}{\end{itemize}\vspace{-4pt}}

% Consistent small bullet for inline items
\newcommand\sbullet[1][.5]{\textbullet}
